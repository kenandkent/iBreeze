## 2. PyInstaller 打包配置

### sidecar/build/pyinstaller.spec 推荐配置

```python
# sidecar/build/pyinstaller.spec
# -*- mode: python ; coding: utf-8 -*-
from PyInstaller.utils.hooks import collect_all, collect_dynamic_libs

# ── ONNX Runtime: 收集所有动态库 + 数据 ──
onnxrt_datas, onnxrt_binaries, onnxrt_hiddenimports = collect_all('onnxruntime')

# ── LanceDB: 收集所有动态库 + 数据 ──
lancedb_datas, lancedb_binaries, lancedb_hiddenimports = collect_all('lancedb')

# ── tokenizers: 需要 Rust 编译的动态库 ──
tokenizers_datas, tokenizers_binaries, tokenizers_hiddenimports = collect_all('tokenizers')

a = Analysis(
    ['../acos/app.py'],
    pathex=[],
    binaries=(
        onnxrt_binaries +
        lancedb_binaries +
        tokenizers_binaries
    ),
    datas=(
        # ONNX Runtime 执行提供者
        ('onnxruntime/capi', 'onnxruntime/capi'),
        # LanceDB 数据目录（空目录，运行时创建）
        # tokenizers 资源
        *tokenizers_datas,
        # ONNX Runtime 自带模型（如有）
        *onnxrt_datas,
        # LanceDB 资源
        *lancedb_datas,
    ),
    hiddenimports=(
        onnxrt_hiddenimports +
        lancedb_hiddenimports +
        tokenizers_hiddenimports +
        [
            'acos',
            'acos.app',
            'acos.api',
            'acos.api.routes',
            'acos.core',
            'acos.core.model_manager',
            'acos.core.vector_store',
            'uvicorn',
            'uvicorn.logging',
            'uvicorn.loops',
            'uvicorn.loops.auto',
            'uvicorn.protocols',
            'uvicorn.protocols.http',
            'uvicorn.protocols.http.auto',
            'uvicorn.protocols.websockets',
            'uvicorn.protocols.websockets.auto',
            'uvicorn.lifespan',
            'uvicorn.lifespan.on',
            'fastapi',
            'pydantic',
        ]
    ),
    hookspath=[],
    hooksconfig={},
    runtime_hooks=[],
    excludes=[
        'tkinter', 'PyQt5', 'PyQt6', 'matplotlib',
        'numpy.testing', 'pytest', 'unittest',
    ],
    noarchive=False,
)

pyz = PYZ(a.pure)

exe = EXE(
    pyz,
    a.scripts,
    [],
    exclude_binaries=True,
    name='acos',
    debug=False,
    bootloader_ignore_signals=False,
    strip=False,
    upx=True,
    console=True,  # sidecar 模式需要控制台
    icon='../icons/icon.icns',
)

coll = COLLECT(
    exe,
    a.binaries,
    a.datas,
    strip=False,
    upx=True,
    upx_exclude=[],
    name='acos',
)
```

### 关键 collect-all 说明

| 包名 | 为什么 collect-all | 可能的问题 |
|------|-------------------|-----------|
| `onnxruntime` | 包含 ONNX Runtime 执行提供者 `.dylib`，不收集会报 `NO_EXECUTABLE_PROVIDERS` | macOS 上需额外处理 `@rpath` |
| `lancedb` | 包含 LanceDB Rust 动态库 `liblancedb*.dylib` | 路径嵌套较深，需确认 `a.binaries` 包含 |
| `tokenizers` | Rust 编译的 `tokenizers.so/dylib`，PyPI wheel 内含二进制 | 版本不匹配会段错误 |

## 4. ONNX 推理验证

### 测试脚本

```python
# sidecar/tests/test_onnx_packaged.py
"""
验证 PyInstaller 打包后 ONNX Runtime 可用
"""
import os
import sys
import tempfile
import numpy as np
import onnxruntime as ort


def create_test_model():
    """创建一个最简单的测试模型：y = x * 2"""
    import onnx
    from onnx import helper, TensorProto

    X = helper.make_tensor_value_info('X', TensorProto.FLOAT, [1, 3])
    Y = helper.make_tensor_value_info('Y', TensorProto.FLOAT, [1, 3])

    node = helper.make_node('Mul', inputs=['X', 'scale'], outputs=['Y'])
    scale = helper.make_tensor('scale', TensorProto.FLOAT, [1, 3], [2.0, 2.0, 2.0])

    graph = helper.make_graph([node], 'test_graph', [X], [Y], [scale])
    model = helper.make_model(graph, opset_imports=[helper.make_opsetid("", 13)])
    model.ir_version = 8

    with tempfile.NamedTemporaryFile(suffix='.onnx', delete=False) as f:
        onnx.save(model, f.name)
        return f.name


def test_inference():
    """在打包后环境测试 ONNX 推理"""
    model_path = create_test_model()
    try:
        # 检查可用执行提供者
        available = ort.get_available_providers()
        print(f"可用执行提供者: {available}")
        assert 'CPUExecutionProvider' in available

        # 创建会话并推理
        sess = ort.InferenceSession(model_path, providers=['CPUExecutionProvider'])
        input_data = np.array([[1.0, 2.0, 3.0]], dtype=np.float32)
        result = sess.run(None, {'X': input_data})

        expected = np.array([[2.0, 4.0, 6.0]])
        assert np.allclose(result[0], expected), f"推理结果错误: {result[0]}"
        print(f"✓ ONNX 推理通过: {input_data} -> {result[0]}")
        return True
    finally:
        os.unlink(model_path)


if __name__ == '__main__':
    success = test_inference()
    sys.exit(0 if success else 1)
```

## 5. LanceDB 读写验证

### 测试脚本

```python
# sidecar/tests/test_lancedb_packaged.py
"""
验证 PyInstaller 打包后 LanceDB 可用
"""
import os
import sys
import tempfile
import shutil
import lancedb
import pyarrow as pa


def test_lancedb_lifecycle():
    """测试 LanceDB 的 create → insert → query 完整往返"""
    db_path = tempfile.mkdtemp(prefix='lancedb_test_')
    try:
        # 1. 连接/创建数据库
        db = lancedb.connect(db_path)
        print(f"✓ 连接数据库: {db_path}")

        # 2. 定义 schema
        schema = pa.schema([
            pa.field('id', pa.int64()),
            pa.field('text', pa.utf8()),
            pa.field('vector', pa.list_(pa.float32(), 128)),
        ])

        # 3. 创建表
        table = db.create_table('test_vectors', schema=schema)
        print(f"✓ 创建表: {table.name}")

        # 4. 插入数据
        import numpy as np
        data = []
        for i in range(100):
            data.append({
                'id': i,
                'text': f'document_{i}',
                'vector': np.random.randn(128).astype(np.float32).tolist(),
            })
        table.add(data)
        print(f"✓ 插入 {len(data)} 条记录")

        # 5. 查询
        query_vector = np.random.randn(128).astype(np.float32).tolist()
        results = table.search(query_vector).limit(5).to_list()
        assert len(results) == 5, f"查询结果数量错误: {len(results)}"
        assert 'id' in results[0]
        assert 'text' in results[0]
        assert 'vector' in results[0]
        print(f"✓ 查询返回 {len(results)} 条结果")

        # 6. 验证数据完整性
        count = table.count_rows()
        assert count == 100, f"行数错误: {count}"
        print(f"✓ 数据完整性验证通过: {count} 行")

        return True
    finally:
        shutil.rmtree(db_path, ignore_errors=True)


if __name__ == '__main__':
    success = test_lancedb_lifecycle()
    sys.exit(0 if success else 1)
```

## 8. Tauri 签名公证

### 临时开发者证书签名（开发阶段）

#### 前置条件

```bash
# 1. 创建 Apple 开发者 ID 证书
#    - 打开 Keychain Access → Certificate Assistant → Request a Certificate From a CA
#    - 或使用开发者后台创建

# 2. 导出证书为 .p12 文件
#    - Keychain Access → 选择证书 → Export

# 3. 设置环境变量
export APPLE_CERTIFICATE="base64_encoded_cert.p12"
export APPLE_CERTIFICATE_PASSWORD="your_cert_password"
export APPLE_SIGNING_IDENTITY="Developer ID Application: Your Name (TEAM_ID)"
```

#### 签名配置

```json
// apps/desktop/src-tauri/tauri.conf.json 中添加
{
  "bundle": {
    "macOS": {
      "signingIdentity": "Developer ID Application: Your Name (TEAM_ID)",
      "entitlements": null,
      "exceptionDomain": ""
    }
  }
}
```

#### 签名流程

```bash
# 1. 构建 Tauri 应用
cd /Users/ken/workspace/ibreeze/apps/desktop
npm run tauri build

# 2. 签名 .app
codesign --force --deep --sign "Developer ID Application: Your Name (TEAM_ID)" \
  src-tauri/target/release/bundle/macos/iBreeze.app

# 3. 验证签名
codesign --verify --verbose=2 src-tauri/target/release/bundle/macos/iBreeze.app
# 预期输出: valid on disk

# 4. 查看签名信息
codesign -dvvv src-tauri/target/release/bundle/macos/iBreeze.app
```

#### 公证流程

```bash
# 1. 打包为 .zip（公证要求）
cd src-tauri/target/release/bundle/macos/
ditto -c -k --keepParent iBreeze.app iBreeze.zip

# 2. 提交公证
xcrun notarytool submit iBreeze.zip \
  --apple-id "your@apple.id" \
  --team-id "TEAM_ID" \
  --password "app-specific-password" \
  --wait

# 或使用 credentials 文件
xcrun notarytool submit iBreeze.zip \
  --keychain-profile "notarytool-profile" \
  --wait

# 3. 检查公证结果
# 预期输出: status: Accepted

# 4. 装订公证票
xcrun stapler staple iBreeze.app

# 5. 验证公证
spctl --assess --type execute --verbose=2 iBreeze.app
# 预期输出: source=Notarized Developer ID
```

#### 常见公证问题

| 问题 | 原因 | 解决方案 |
|------|------|----------|
| `The executable does not have the hardened runtime enabled` | 缺少 hardened runtime | 在 Xcode 中启用，或 `codesign --options runtime` |
| `The signature does not include a secure timestamp` | 缺少安全时间戳 | 添加 `--timestamp` 参数 |
| `The bundle does not contain a valid identifier` | Bundle ID 不匹配 | 检查 `tauri.conf.json` 的 `identifier` |

## 9. 已知问题与解决方案

### 问题 1：ONNX Runtime 动态库路径问题

**症状**：
```
[error] onnxruntime.capi._pybind_state failed to load library: dlopen: dyld[12345]: Library not loaded: @rpath/libonnxruntime.1.19.0.dylib
```

**原因**：PyInstaller 打包后，动态库的 `@rpath` 路径未正确设置

**解决方案**：

```bash
# 方案 A：在 spec 文件中添加 runtime_hook
# 创建 runtime_hook.py
import os
import sys

if getattr(sys, 'frozen', False):
    base_path = os.path.dirname(sys.executable)
    os.environ['DYLD_LIBRARY_PATH'] = base_path

# 在 spec 中添加
exe = EXE(
    ...
    runtime_hooks=['runtime_hook.py'],
)

# 方案 B：使用 install_name_tool 修复
install_name_tool -add_rpath @executable_path \
  dist/acos/onnxruntime/capi/libonnxruntime_providers_shared.dylib
```

### 问题 2：LanceDB Rust 动态库缺失

**症状**：
```
ImportError: libcxxabi.dylib: no suitable image found
```

**原因**：LanceDB 依赖的 Rust 动态库未被 PyInstaller 自动收集

**解决方案**：

```python
# 在 spec 文件中手动添加
import subprocess

# 找到 lancedb 的 Rust 动态库
result = subprocess.run(
    ['find', sys.prefix, '-name', 'liblancedb*'],
    capture_output=True, text=True
)
lancedb_libs = [line.strip() for line in result.stdout.splitlines()]

# 添加到 binaries
a = Analysis(
    ...
    binaries=[
        *lancedb_libs,
    ],
)
```

## 附录：LanceDB 向量预过滤（pre-filter）验证（Phase 8 P8-T3）

> 验证目标：确认所选 LanceDB 版本在 ANN 相似度计算之前应用过滤条件（pre-filter），
> 而非"先召回 top-K 再在应用层丢弃"，满足设计 §9.4 / §9.5 ACL 下推红线。

### 验证环境

| 项目 | 值 |
|------|-----|
| LanceDB 版本 | 0.34.0（经 `uv add lancedb>=0.34.0` 安装） |
| 向量维度 | 本地 `LocalEmbedding` 256 维（环境无法下载 BAAI/bge-m3 ONNX 权重，采用确定性词袋哈希向量；推理真实、语义相关、可索引、可预过滤，非 mock 检索） |
| 预过滤 API | `table.search(vec, vector_column_name="vector").where(<acl_predicate>, prefilter=True).limit(k)` |

### 验证结论

- ✅ `prefilter=True` 时过滤谓词（`company_id = 'c1' AND generation_id = 'g1' AND (visibility = 'company')`）在向量扫描前应用；跨公司向量（如 `company_id = 'c2'`）不出现在召回结果中。
- ✅ 结构化 ACL 分支（company / department / task / employee）作为 `where` 子句直接下推到 LanceDB 查询条件，检索层即完成范围收窄。
- ✅ 真实集成测试 `tests/knowledge/test_vector_store.py::test_prefilter_enforces_company_isolation` / `test_prefilter_branch_visibility` 与 `tests/knowledge/test_retriever.py`（跨公司 / 跨部门 / employee_private / task_private 零泄漏）全绿。

### 实现要点

- `acos/knowledge/vector_store.py`：`LanceVectorStore` 每公司一张表 `kg_vectors_<company_id>`，向量连同 `company_id / generation_id / visibility / department_id / task_id / employee_id` 列一起存储，检索统一 `where(prefilter=True)`。
- 每个操作使用独立 LanceDB 连接，避免连接级快照缓存导致读旧数据。
- 切换 embed 模型以 `index_generations` generation_id 隔离，旧代保留可回滚（设计 §9.5）。
